//selection

#include <iostream>
#include <vector>
#include <chrono>

using namespace std;
using namespace std::chrono;

struct OperationCount {
    int comparisons = 0;
    int assignments = 0;

    void reset() {
        comparisons = 0;
        assignments = 0;
    }

    void print() const {
        cout << "Operation Count:" << endl;
        cout << "Comparisons: " << comparisons << endl;
        cout << "Assignments: " << assignments << endl;
    }

    int total() const {
        return comparisons + assignments;
    }
};

void selectionSort(vector<int>& arr, OperationCount& opCount) {
    int n = arr.size();
    for (int i = 0; i < n - 1; i++) {
        int min_idx = i;
        for (int j = i + 1; j < n; j++) {
            opCount.comparisons++;
            if (arr[j] < arr[min_idx]) {
                min_idx = j;
                opCount.assignments++;
            }
        }
        if (min_idx != i) {
            swap(arr[i], arr[min_idx]);
            opCount.assignments += 3; // Swap operation involves three assignments
        }
    }
}

int main() {
    int n;
    cout << "Enter the number of elements: ";
    cin >> n;

    vector<int> arr(n);
    cout << "Enter the elements: ";
    for (int i = 0; i < n; i++)
        cin >> arr[i];

    OperationCount opCount;

    auto start = high_resolution_clock::now(); // Start time
    selectionSort(arr, opCount);
    auto stop = high_resolution_clock::now(); // End time

    auto duration = duration_cast<microseconds>(stop - start);

    cout << "Sorted array: ";
    for (const int& num : arr)
        cout << num << " ";
    cout << endl;

    cout << "Time taken by function: " << duration.count() << " microseconds" << endl;
    opCount.print();
    cout << "Total operations: " << opCount.total() << endl;

    return 0;
}
